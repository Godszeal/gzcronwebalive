Beginner-friendly step-by-step

1) Create project: npm init -y
2) Install: npm i gzcronwebalive
3) Add to index.js:
// index.js
const { startCron, events } = require('gzcronwebalive');
startCron();
events.on('log', msg => console.log('CRON LOG:', msg));
4) Run: node index.js

Example server (SSE + docs)

Save the provided example_server.js and place this index.html under docs/. Start server with node example_server.js ‚Äî open http://localhost:3000/docs/index.html.

// example_server.js
const express = require('express');
const http = require('http');
const path = require('path');
const { startCron, events } = require('./gzcronwebalive'); // local module copy

const app = express();
const server = http.createServer(app);
app.use('/docs', express.static(path.join(__dirname, 'docs')));

let clients = [];
app.get('/events', (req, res) => {
  res.setHeader('Content-Type','text/event-stream');
  res.setHeader('Cache-Control','no-cache');
  res.setHeader('Connection','keep-alive');
  res.flushHeaders();
  clients.push(res);
  req.on('close', () => { clients = clients.filter(c => c !== res); });
});

events.on('log', msg => {
  clients.forEach(res => res.write(`data: ${msg}\n\n`));
});

startCron(); // start the cron
server.listen(process.env.PORT||3000, () => console.log('Server listening on port', process.env.PORT||3000));
